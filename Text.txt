Function UtilOneLineAddress(rs)
	Dim s, v
	s = rs("Address_Line_1").Value
	If rs("Address_Line_2").Value & "" <> "" Then
		s = s & ", " & rs("Address_Line_2").Value
	End If
	If rs("Address_Line_3").Value & "" <> "" Then
		s = s & ", " & rs("Address_Line_3").Value
	End If
	If rs("City").Value & "" <> "" Then
		s = s & ", " & rs("City").Value
	End If
	If rs("State_Code").Value & "" <> "" Then
		s = s & ", " & rs("State_Code").Value
	End If
	If rs("Zip_Code").Value & "" <> "" Then
		s = s & " " & rs("Zip_Code").Value
	End If
	If rs("Country_Name").Value & "" <> "" Then
		s = s & ", " & rs("Country_Name").Value
	End If
	UtilOneLineAddress = s
End Function
